encodingMap = {
	'utf8': 'utf8',
	'utf-8': 'utf8',

	'latin1': 'latin1',
	'latin-1': 'latin1',
	'iso-8859-1': 'latin1',
	'iso8859-1': 'latin1',

module.exports.toUtf8 = (text, encoding) ->
	if encoding? and text?
		encoding = encodingMap[encoding.toLowerCase()]
		if encoding?
			str = iconv.fromEncoding(new Buffer(text, 'binary'), encoding)
			text = iconv.toEncoding(str, 'utf8').toString()
	return text
